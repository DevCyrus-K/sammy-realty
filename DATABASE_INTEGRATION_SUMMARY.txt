1. LOGIN EXAMPLE
   const { email, password } = formData;
   const response = await fetch('/api/admin/login', {
     method: 'POST',
     body: JSON.stringify({ email, password })
   });
   // Returns: { success: true, user: { id, email } }

2. PROPERTIES EXAMPLE
   const { properties, loading } = useProperties({ featured: true });
   // Automatically fetches from /api/properties?featured=true

3. SUBMIT INQUIRY EXAMPLE
   const { submitInquiry } = useInquiry();
   await submitInquiry({
     name: 'John Doe',
     email: 'john@example.com',
     message: 'Interested in property',
     propertyId: 1
   });
